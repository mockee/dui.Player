dui.Player ♪

a simple HTML5 audio player.

How to use?

douban style:

        <!-- include the css and scripts in your document -->
        <link rel="stylesheet" type="text/css" href="css/player.css" /> 
        <script type="text/javascript" src="audio.js"></script>
        <script type="text/javascript" src="player.js"></script>
        
 
        <!-- target div -->
        <div id="audio"></div> 
        
        // select the div
        var audio = document.getElementById('audio'),
            
            // prepare songs list
            songsList = [
                {
                    name: 'People Are Strange',
                    artist: 'The Doors',
                    url: 'mp3/people_are_strange.mp3'
                },
                {
                    name: 'Last Flowers',
                    artist: 'Radiohead',
                    url: 'mp3/last_flowers.mp3'
                },
                {
                    name: '虹が始まるとき',
                    artist: '告白 オリジナル・サウンドトラック',
                    url: 'mp3/g.mp3'
                },
                {
                    name: 'Hold My Hand',
                    artist: 'Micheal Jackson',
                    url: 'mp3/hold_my_hand.mp3'
                }

            ],

            // init player
            player = dui.Player(audio);

        // init playlist 
        player.init(songsList);

        // create songs list
        player.createList();

Native style:

        <!-- target div -->
        <div id="audio2"></div>
        
        // controls: true
        var nativePlayer = dui.Radio(audio2, true);

        // init playlist
        nativePlayer.init(songsList);
